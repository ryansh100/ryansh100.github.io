    import {Request} from '../request.es6';

/**
 * Represents a known color object.
 */
export class KnownColors {

  /**
   * Method to get a known
   * @param {string} type Catalog item type
   * @param {string} number Catalog item number
   */
  static get(type, number) {
    let uri = `/items/${type}/${no}/color`;

    return new Request(uri, [], (data) => {
      return data.map(color => new KnownColors(color));
    });
  }
}
